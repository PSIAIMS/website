# ---- Robust package setup ----

options(repos = c(CRAN = "https://cran.rstudio.com"))

required_packages <- c(
  "shiny",
  "dplyr",
  "teal"
)

installed_packages <- rownames(installed.packages())
missing_packages <- setdiff(required_packages, installed_packages)

if (length(missing_packages) > 0) {
  install.packages(
    missing_packages,
    type = "binary",
    dependencies = TRUE
  )
}

library(shiny)
library(dplyr)
library(teal)

data <- teal_data(
  IRIS = iris,
  MTCARS = mtcars
)

app <- init(
  data = data,
  modules =example_module()) %>%  

  modify_title("Basic Teal Demo App") %>% 
  modify_header("My first teal app") %>% 
  modify_footer("Powered by teal + shiny")


shinyApp(app$ui, app$server)
